```
<?php 
    $ch = curl_init(); 
    curl_setopt($ch, CURLOPT_URL, "https://explorer.tyzen.io/api/getconnectioncount");
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1); 
    $output = curl_exec($ch); 
    curl_close($ch);      
    echo $output;
?>
```
